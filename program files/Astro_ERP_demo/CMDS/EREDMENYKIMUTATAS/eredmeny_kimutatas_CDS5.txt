  (Select coalesce(avg(be_ar_netto),0) from arukeszlet where arukeszlet.arutorzs_id=AT.id and arukeszlet.keszlet<>0 and arukeszlet.raktar_id in (select id from raktar where tipus=0 and archiv=0) )
  *
  ((afa.kulcs / 100 )+1)
  *
  (Select coalesce(sum(keszlet),0) from arukeszlet where arukeszlet.arutorzs_id=AT.id and arukeszlet.keszlet<>0 and arukeszlet.raktar_id in (select id from raktar where tipus=0 and archiv=0) ) 
  as ossz_beszerzesi_ar
  
  FROM arutorzs AT, afa  WHERE afa.id=AT.afa_id  HAVING keszlet<>0  ORDER BY AT.teljes_megnevezes
    
  ) T 
) normal, /* ahol a raktár típusa normál és nem archiv */
