select distinct penztar_id, bankszamla_id, penztarnev, banknev, tipus, 
case
  when penztar_id is not null then penztarnev
  when bankszamla_id is not null then banknev
end nev, concat(concat(concat(tipus,coalesce(penztarnev,'')),coalesce(banknev,'')),coalesce(bankszamlaszam,'')) nevtype,
sum(osszeg * arfolyam) vosszeg,
b.bankszamlaszam
from 
penzmozgasok pm 
left join penztar p on p.id = pm.penztar_id
left join bankszamla b on b.id = pm.bankszamla_id
where pm.kelte BETWEEN {1} AND {2}
group by nevtype, nev + bankszamlaszam
order by tipus
