bhe.hatralek_ft szla_osszeg, bhe.hatralek, bhe.ossz_brutto,
CASE
 WHEN bh.hatarido<Curdate() THEN 'LEJ¡RT A HAT¡RID’!'

END allapot
      
      FROM partner p  
      left join biz_header bh on bh.partner_id = p.id
