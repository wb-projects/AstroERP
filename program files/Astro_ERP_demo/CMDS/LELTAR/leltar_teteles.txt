SELECT arukeszlet.id, arukeszlet.kod, arukeszlet.egyedi_azonosito, arukeszlet.egyedi_megnevezes,
arukeszlet.beszerzesi_ar, arukeszlet.keszlet, arukeszlet.arutorzs_kod, arukeszlet.beszerzesi_ar,
arukeszlet.keszlet as k_mennyi,
round(arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1)) as netto_eladar,
round(arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1)) as netto_ertek,
round(((arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1))*((afa.afakulcs/100)+1))+0.0001) as brutto_eladar,
round(((arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1))*((afa.afakulcs/100)+1))+0.0001)*arukeszlet.keszlet as k_eladar,
arutorzs.arutorzs_nev, arutorzs.arutorzs_ksh, arutorzs.m_egys,
arucsop.arucsop_jelleg,
afa.afakulcs,
leltar.l_mennyi,
round(((arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1))*((afa.afakulcs/100)+1))+0.0001)*leltar.l_mennyi as l_eladar,
(leltar.l_mennyi - arukeszlet.keszlet) as ht_mennyi,
round(((arukeszlet.beszerzesi_ar*((arutorzs.ar_szorzo{1}/100)+1))*((afa.afakulcs/100)+1))+0.0001)*(leltar.l_mennyi - arukeszlet.keszlet) as ht_eladar,
leltar.megjegyzes
FROM arukeszlet,arutorzs,arucsop,afa,leltar
WHERE arutorzs.kod=arukeszlet.arutorzs_kod
AND arucsop.kod=arutorzs.arucsop_kod
AND afa.kod=arutorzs.afa_kod
AND leltar.arukeszlet_kod=arukeszlet.kod
AND ( (leltar.l_mennyi <> arukeszlet.keszlet) OR (arukeszlet.keszlet>0) ) 
ORDER BY arutorzs.arutorzs_nev, arukeszlet.egyedi_megnevezes, arukeszlet.egyedi_azonosito
