SELECT
str_to_date('{1}', '%Y.%m.%d.' ) as Date1, str_to_date('{2}', '%Y.%m.%d.' ) AS Date2,
hoz.beszallito_kod AS ugyfelkod, besz.nev AS ugyfelnev, hoz.arukeszlet_kod,
arukeszlet.egyedi_azonosito, arutorzs.arutorzs_nev, arukeszlet.egyedi_megnevezes,
(select sum(mennyiseg) from hozzarendeles where mennyiseg>0 and beszallito_kod=hoz.beszallito_kod and arukeszlet_kod=hoz.arukeszlet_kod and datum between Date1 and Date2) AS kiadott,
(select abs(sum(mennyiseg)) from hozzarendeles where mennyiseg<0 and beszallito_kod=hoz.beszallito_kod and arukeszlet_kod=hoz.arukeszlet_kod and datum between Date1 and Date2) AS eladott,
sum(mennyiseg) as kintlevo,
arukeszlet.eladasi_ar AS netto_egysegar,
(arukeszlet.eladasi_ar * (1+(afa.afakulcs/100)) ) as brutto_egysegar,
(select eladott*netto_egysegar) AS netto_ertek,
(select eladott*brutto_egysegar) AS brutto_ertek,
(select brutto_ertek-netto_ertek) AS afa_ertek
FROM hozzarendeles AS hoz, partner AS besz, arukeszlet, arutorzs, afa
WHERE besz.kod=hoz.beszallito_kod
AND arukeszlet.kod=hoz.arukeszlet_kod
AND arutorzs.kod=arukeszlet.arutorzs_kod
AND afa.kod=arutorzs.afa_kod
and datum between (select Date1) and (select Date2)
GROUP BY hoz.beszallito_kod, hoz.arukeszlet_kod
