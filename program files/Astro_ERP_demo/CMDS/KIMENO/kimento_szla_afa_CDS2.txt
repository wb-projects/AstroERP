    SELECT
     bh.id,
     sorszam,
     partner_nev,
     fizmod,
     CASE
      WHEN storno = 0 AND sum(bie.ossz_brutto_ft) <= 0 THEN bh.megjegyzes
      WHEN storno = 1 THEN 'Stornů'
      WHEN bh.hatarido < NOW() AND CalcHatralek(bh.id)>0 AND bh.fizmod <> 'K…SZP…NZ' THEN 'LejŠrt a fizetťsi hatŠridű !'
      WHEN bh.hatarido >= NOW() AND CalcHatralek(bh.id)>0 AND bh.fizmod <> 'K…SZP…NZ' THEN 'Tartozik'
      ELSE 'Kifizetve'
     END allapot,

     sum(bie.ossz_netto_ft) as ossz_netto,
     sum(bie.ossz_afa_ft) as ossz_afa,
     sum(bie.ossz_brutto_ft) as ossz_brutto
 
    FROM biz_header bh, biz_header_ext bhe, biz_item bi, biz_item_ext bie 
    WHERE
     bh.id = bhe.biz_header_id 
    AND
     bi.biz_header_id = BH.id 
    AND 
     bie.biz_item_id = bi.id
    AND
     kelte BETWEEN {1} AND {2} {3}
    AND
     biz_modul_kod = {4} 
    AND 
     bh.lezarva = 1 
    GROUP BY bh.id
    ORDER BY kelte
