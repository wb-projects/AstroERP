SELECT
W.actual_week, 
Max(M.production_unit) AS production_unit,
Sum(ML.working_time/60) AS working_time_week,
Sum(ML.produced_quantity) AS produced_quantity_week,
Sum(ML.working_time/60)/Sum(ML.produced_quantity) AS productivity_week,
Max(M.production_target_week) AS production_target_week,
( (Sum(ML.working_time/60)/Sum(ML.produced_quantity)) / Max(M.production_target_week) ) * 100 AS efficiency_week,
100
FROM 

(select (t*10+u+1) actual_week from
(select 0 t union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) A,
(select 0 u union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) B
where (t*10+u+1)<53
order by actual_week) W

LEFT OUTER JOIN machines M ON M.id=:machine_id
LEFT OUTER JOIN machine_log ML ON ML.machine_id=M.id AND W.actual_week=WEEK(ML.work_date,7) AND YEAR(ML.work_date)=:year
GROUP BY W.actual_week
ORDER BY W.actual_week
