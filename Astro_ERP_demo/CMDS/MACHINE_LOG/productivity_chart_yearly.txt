SELECT
Y.actual_year AS actual_year, 
Max(M.production_unit) AS production_unit,
Sum(ML.working_time/60) AS working_time_year,
Sum(ML.produced_quantity) AS produced_quantity_year,
Sum((ML.working_time/60)/ML.produced_quantity) AS productivity_year,
Max(M.production_target_year) AS production_target_year,
( (Sum(ML.working_time/60)/Sum(ML.produced_quantity)) / Max(M.production_target_year) ) * 100 AS efficiency_year,
100
FROM 

(select (th*1000+h*100+t*10+u+1) actual_year from
(select 0 th union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) A,
(select 0 h union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) B,
(select 0 t union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) C,
(select 0 u union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) D
where (th*1000+h*100+t*10+u+1) between :year - 10 and :year
order by actual_year) Y

LEFT OUTER JOIN machines M ON M.id=:machine_id
LEFT OUTER JOIN machine_log ML ON ML.machine_id=M.id AND Y.actual_year=YEAR(ML.work_date)
GROUP BY Y.actual_year
ORDER BY Y.actual_year
