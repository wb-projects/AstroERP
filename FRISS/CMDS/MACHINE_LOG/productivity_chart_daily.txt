SELECT
D.actual_date, Day(D.actual_date) AS actual_day, ML.id, 
M.production_unit,
ML.working_time/60 AS working_time_day,
ML.produced_quantity AS produced_quantity_day,
((ML.working_time/60)/ML.produced_quantity) AS productivity_day,
M.production_target_day,
( ((ML.working_time/60)/ML.produced_quantity) / M.production_target_day ) * 100 AS efficiency_day,
100
FROM 

(select gen_date AS actual_date from 
(select adddate('1970-01-01',t4*10000 + t3*1000 + t2*100 + t1*10 + t0) gen_date from
 (select 0 t0 union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) t0,
 (select 0 t1 union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) t1,
 (select 0 t2 union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) t2,
 (select 0 t3 union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) t3,
 (select 0 t4 union select 1 union select 2 union select 3 union select 4 union select 5 union select 6 union select 7 union select 8 union select 9) t4) v
where year(gen_date)=:year and month(gen_date)=:month) D

LEFT OUTER JOIN machines M ON M.id=:machine_id
LEFT OUTER JOIN machine_log ML ON ML.machine_id=M.id AND D.actual_date=ML.work_date
ORDER BY D.actual_date
