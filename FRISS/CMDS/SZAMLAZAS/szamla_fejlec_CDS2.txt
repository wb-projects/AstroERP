SELECT biz_header.*, biz_header_ext.*,
(Select If(egyeni_vallalkozo=1,ev_nyilv_tart_szam,ceg_engszam) From settings Limit 1) as eng_nyilv_szam,
(Select cim From telephely Where id=biz_header.telephely_id) as telephely_cim,
(Select telfax From telephely Where id=biz_header.telephely_id) as telephely_tel,
(Select email From partner Where id=biz_header.partner_id) as partner_email,
(Select GROUP_CONCAT(DISTINCT BIE.source_item_header_serial ORDER BY BI.id DESC SEPARATOR ', ') 
From biz_item BI, biz_item_ext BIE Where BI.id=BIE.biz_item_id And BI.biz_header_id=biz_header.id 
And BIE.source_item_header_serial is not null) as items_source_biz_list
FROM biz_header, biz_header_ext 
WHERE biz_header_ext.biz_header_id=biz_header.id AND biz_header.id='{1}'
