SELECT
  'ITEM' as itipus, biz_item.id, extra_megjegyzes, megnevezes, megjegyzes, media_deletion_code, ktdij_text, afa_text, (select vtsz from arutorzs where id=biz_item.arutorzs_id) as vtsz,
  mennyiseg, egyseg, egysegar, engedmeny,
  ossz_netto_alap as ertek_netto,
  afakulcs as afa, afanev as afa_nev,
  ossz_afa_alap as afa_osszeg,
  ossz_brutto_alap as ertek_brutto,
  band,
  egysegar_eng as e_egysegar,
  ossz_netto_eng as e_ertek_netto,
  ossz_afa_eng as e_afa_osszeg,
  ossz_brutto_eng as e_ertek_brutto,
  egysegar_csk as c_egysegar,
  ossz_netto as c_ertek_netto,
  ossz_afa as c_afa_osszeg,
  ossz_brutto as c_ertek_brutto
  FROM biz_item, biz_item_ext
  WHERE biz_item_ext.biz_item_id=biz_item.id AND biz_header_id='{1}'

  UNION

  SELECT DISTINCT
  'AFA', 0, '-', IF(LENGTH(afa.megnevezes)<4,'',CONCAT(afa.megnevezes,':')), '-', '', '', '', '',
  '-', '-', '-', 0,
  sum(ossz_netto) as ertek_netto,
  afakulcs as afa, afanev as afa_nev,
  sum(ossz_afa) as afa_osszeg,
  sum(ossz_brutto) as ertek_brutto,
  band,
  '-' as e_egysegar,
  '-' as e_ertek_netto,
  '-' as e_afa_osszeg,	
  '-' as e_ertek_brutto,
  '-' as c_egysegar,
  '-' as c_ertek_netto,
  '-' as c_afa_osszeg,
  '-' as c_ertek_brutto
  FROM biz_item, biz_item_ext, afa
  WHERE biz_item_ext.biz_item_id=biz_item.id AND afa.id=afa_id AND biz_header_id='{1}'
  GROUP BY band, afa_id

  UNION

  SELECT DISTINCT
  'SUM', 0, '-', '-', '-', '', '', '', '',
  '-', '-', '-', 0,
  sum(ossz_netto) as ertek_netto,
  '-', '-',
  sum(ossz_afa) as afa_osszeg,
  sum(ossz_brutto) as ertek_brutto,
  band,
  '-' as e_egysegar,
  '-' as e_ertek_netto,
  '-' as e_afa_osszeg,
  '-' as e_ertek_brutto,
  '-' as c_egysegar,
  '-' as c_ertek_netto,
  '-' as c_afa_osszeg,
  '-' as c_ertek_brutto
  FROM biz_item, biz_item_ext
  WHERE biz_item_ext.biz_item_id=biz_item.id AND biz_header_id='{1}'
  GROUP BY band


  ORDER BY ifnull(band,''), FIELD(itipus,'ITEM','AFA','SUM'), id
