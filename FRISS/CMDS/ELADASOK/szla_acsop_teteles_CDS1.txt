SELECT ACS.id AS arucsop_id, ACS.megnevezes as arucsop_nev, Sum(SZI.mennyiseg) AS mennyiseg,
Sum(Round(SZIE.ossz_brutto_ft)) AS ertek_brutto
FROM biz_header SZH, biz_item SZI, biz_item_ext SZIE, arutorzs A, arucsop ACS
WHERE SZI.biz_header_id=SZH.id
AND SZI.id=SZIe.biz_item_id
AND A.id=SZI.arutorzs_id
AND ACS.id=A.arucsop_id
AND A.tipus='A'
AND SZH.biz_modul_kod=1301
AND SZH.sorszam not like 'UJ%'
AND SZH.kelte BETWEEN {1} AND {2}
GROUP BY ACS.id
ORDER BY ACS.megnevezes
