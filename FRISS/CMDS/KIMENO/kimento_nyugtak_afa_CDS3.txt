    SELECT DISTINCT szamla_items.szamla_szam, szamla_items.afa, 
     sum(szamla_items.ertek_netto-( (szamla_items.engedmeny/100)*szamla_items.ertek_netto)) as a_ertek_netto, 
     sum(szamla_items.afa_osszeg-( (szamla_items.engedmeny/100)*szamla_items.afa_osszeg)) as a_afa_osszeg, 
     sum(szamla_items.ertek_brutto-( (szamla_items.engedmeny/100)*szamla_items.ertek_brutto)) as a_ertek_brutto FROM szamla_header, szamla_items 
     WHERE szamla_items.szamla_szam=szamla_header.szamla_szam AND szamla_header.kelte BETWEEN {1} AND {2}
      AND szamla_header.tipus LIKE 'NYUGTA' GROUP BY szamla_items.szamla_szam, szamla_items.afa ORDER BY szamla_items.szamla_szam, szamla_items.afa
