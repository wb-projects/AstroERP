SELECT DISTINCT 
 tipus,
 tipus as fotipus,
 biz_modul_kod,
 sum(szla_ertek_brutto) as ossz_brutto,
 
 ( SELECT sum(szla_ertek_brutto) FROM szamla_header_except_mgr_view
 WHERE 1=1
 AND date(kelte)='{2}'
 AND telephely_id={1}
 AND lezarva=1
 AND fizmod='KÉSZPÉNZ'
 AND storno<>1
 AND biz_modul_kod=SHEMG.biz_modul_kod
 ) as kp_ossz,

 ( SELECT sum(szla_ertek_brutto) FROM szamla_header_except_mgr_view
 WHERE 1=1
 AND date(kelte)='{2}'
 AND telephely_id={1}
 AND lezarva=1
 AND fizmod<>'KÉSZPÉNZ'
 AND storno<>1
 AND biz_modul_kod=SHEMG.biz_modul_kod
 ) as utal_ossz

 FROM szamla_header_except_mgr_view AS SHEMG
 WHERE 1=1
 AND date(kelte)='{2}'
 AND telephely_id={1}
 AND lezarva=1
 AND storno<>1 
 GROUP BY biz_modul_kod
 ORDER BY biz_modul_kod DESC ;
