SELECT DISTINCT
 biz_header.id as biz_header_id,
 biz_header.sorszam as szamla_szam,
 biz_item_ext.megnevezes,
 biz_item.engedmeny,
 biz_item.mennyiseg,
 biz_item.egyseg,
 biz_item_ext.ossz_brutto_ft as ossz_ertek_brutto
 FROM biz_header, biz_item, biz_item_ext
 WHERE biz_item.biz_header_id=biz_header.id AND biz_item_ext.biz_item_id=biz_item.id
 AND (biz_item.munkaszam NOT LIKE 'MGR%')
 AND date(biz_header.kelte)='{2}'
 AND telephely_id={1}
 AND biz_header.lezarva=1
 AND biz_header.storno<>1 
 ORDER BY biz_header.sorszam, biz_item.id DESC;
