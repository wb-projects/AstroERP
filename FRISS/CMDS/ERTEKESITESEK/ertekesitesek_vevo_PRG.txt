SELECT hoz.arukeszlet_kod,
arukeszlet.egyedi_azonosito, arukeszlet.egyedi_megnevezes, arukeszlet.eladasi_ar AS netto_egysegar,
arutorzs.arutorzs_nev, arutorzs.kod AS arutorzs_kod,
abs(sum(mennyiseg)) AS mennyi,
(abs(sum(mennyiseg))*arukeszlet.eladasi_ar) AS netto_ertek,
afa.afanev, afa.afakulcs,
(arukeszlet.eladasi_ar * (1+(afa.afakulcs/100)) ) as brutto_egysegar,
(((abs(sum(mennyiseg))*arukeszlet.eladasi_ar)/100)*afa.afakulcs)  AS afa_ertek,
((abs(sum(mennyiseg))*arukeszlet.eladasi_ar)*(1+(afa.afakulcs/100))) AS brutto_ertek
FROM hozzarendeles AS hoz, partner, arukeszlet, arutorzs, afa
WHERE partner.kod=hoz.vevo_kod
AND arukeszlet.kod=hoz.arukeszlet_kod
AND arutorzs.kod=arukeszlet.arutorzs_kod
AND afa.kod=arutorzs.afa_kod
AND szamlazva=0
AND {1}
AND hoz.vevo_kod={2}
GROUP BY hoz.arukeszlet_kod
